How to integrate into React / Next.js / Vue

LaptopHero.tsx · LaptopHero.vue 
// ─────────────────────────────────────────────────────────────────
// REACT / NEXT.JS  (TypeScript)
// Copy the CSS from the <style> block into LaptopHero.module.css
// ─────────────────────────────────────────────────────────────────
import { useEffect, useRef } from 'react'
import styles from './LaptopHero.module.css'

export default function LaptopHero() {
  const screenRef = useRef<HTMLDivElement>(null)
  const lidRef    = useRef<HTMLDivElement>(null)

  const playAnimation = () => {
    const screen = screenRef.current
    const lid    = lidRef.current
    if (!screen || !lid) return

    // ① Reset all classes
    screen.classList.remove('screen-on')
    lid.classList.remove('glowing')
    document.querySelectorAll('.anim-item')
      .forEach(el => el.classList.remove('animate'))

    // ② Force reflow so CSS transitions restart cleanly
    void screen.offsetWidth

    // ③ Screen brightens (CSS transition handles easing)
    setTimeout(() => screen.classList.add('screen-on'), 80)

    // ④ Frame glow (CSS animation on .glowing)
    setTimeout(() => lid.classList.add('glowing'), 300)

    // ⑤ Stagger content — .delay-* classes control per-item timing
    setTimeout(() => {
      document.querySelectorAll('.anim-item')
        .forEach(el => el.classList.add('animate'))
    }, 650)
  }

  useEffect(() => { setTimeout(playAnimation, 400) }, [])

  return (
    <div className={styles.laptopWrap}>
      <div className={styles.laptopLid} ref={lidRef}>
        <div className={styles.laptopScreen} ref={screenRef}>
          {/* ... same HTML structure ... */}
        </div>
      </div>
    </div>
  )
}
